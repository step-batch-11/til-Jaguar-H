wrote a code for simple animation :

```
let balls = []
class ball {
  
  constructor(x,y,size,dx,color){
    this.x = x;
    this.y = y;
    this.size = size
    this.dx = dx
    this.a = size * 0.001
    this.color = color
  }
  
 update (){
   this.x = (this.x + this.dx)%width
   this.y = (this.y + this.a)%height
   this.dx += this.a
   
 }
  
  draw (){
    circle(this.x,this.y,this.size)
  }
}

function setup() {
  createCanvas(800, 800);
  
  for (let i = 0 ; i < 400; i++){
    console.log(i)
    let x = random(1,width)
    let y = random(1,height)
    let [size,dx] = random([[2,0],[5,0.1],[3,0.05]])
    const color = random(["red","blue","green","yellow"])
    balls.push(new ball(x,y,size,dx,color))
  }
}

function draw() {
  background(1,1,1,20);
  noStroke()
  

  for (const ball of balls){
    fill(ball.color)
    ball.draw()
    ball.update()
  }
}
```
